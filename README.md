source ~/.bash_profile

nvm alias default v9.8.0

安装libpcap
yum -y install gcc-c++

yum -y install flex

yum -y install bison

wget http://www.tcpdump.org/release/libpcap-1.8.1.tar.gz

tar -zxvf libpcap-1.8.1.tar.gz

cd libpcap-1.8.1

./configure

make

make install
